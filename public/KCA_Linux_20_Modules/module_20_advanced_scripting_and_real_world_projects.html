Error Handling

#!/bin/bash set -euo pipefail trap 'echo Error on line $LINENO' ERR set -e: Exit on error set -u: Error on undefined vars set -o pipefail: Catch pipe errors trap: Run command on signal

System Health Script

#!/bin/bash echo '=== System Health ===' echo "CPU Load: $(uptime | awk -F'load average:' '{print $2}')" echo "Memory: $(free -h | awk '/Mem/{print $3"/"$2}')" echo "Disk: $(df -h / | awk 'NR==2{print $5}') used" echo "Uptime: $(uptime -p)"

Backup Script

#!/bin/bash SOURCE='/var/www' DEST='/backup' DATE=$(date +%Y%m%d) tar -czf $DEST/backup-$DATE.tar.gz $SOURCE [ $? -eq 0 ] && echo 'Backup OK' || echo 'Backup FAILED'

User Provisioning

while IFS=',' read -r username email; do id $username &>/dev/null && echo "$username exists" || { useradd -m $username echo "Created: $username" } done < users.csv
